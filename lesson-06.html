Iterations in JavaScript

prejdenie prvkov poľa


// ES5 Array
var arr = ['a', 'b', 'c'];
for (var i = 0; i < arr.length; i++) {
    console.log(i + ': ' + arr[i]);
}

prejdenie prvkov poľa


// ES5/6
arr.forEach((item, index) => {
    console.log(`${index}: ${item}`);
})
// ES6
for (let item of arr) {
    console.log(item);
}

prejdenie prvkov objektu


// ES5 Object
var obj = {
    id: 10,
    name: 'Adam'
};
for (var key in obj) {
    console.log(key + ' = ' + obj[key]);
}

prejdenie prvkov objektu


// ES5/6
Object.keys(obj).forEach((key) => {
    console.log(`${key}: ${obj[key]}`);
});
